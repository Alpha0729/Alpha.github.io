ajax数据交互

1.AJAX 是与服务器交换数据的技术，它在不重载全部页面的情况下，实现了对部分网页的更新。

get:容量小（4k-10k）不适合密码提交,不适合大文件传输,适合用于网页分享,get是有本地缓存的

post:容量大(2G)，相对get,比较安全,post是没有缓存的

2.jquery的get(url,fn)方法：url:请求的API,fn：请求成功时回调的函数

        $(document).ready(function () {
            $("#btn").click(function () {
                //清除本地缓存："../text/1.text/？t="+new Date().getTime()
                //$.get('../text/1.text', function (data,status) {
                //data:请求的数据;status:成功状态
                $.get('../text/1.text', function (data,status) {
                alert("请求到的数据："+data+":成功状态"+status)
                });
            })
        });

3.jquery的post(url,data,fn):url:请求地址;data:要传送的数据;fn：成功回调函数

            $(document).ready(function () {
                $("#btn2").click(function () {
                    $.post("../text/1.php",{name:"liu",value:'18'}, function (data,status) {
                        alert(data+":"+status)
                    });
                });
            });

4.angular.js的get/post通用方法：

            var app = angular.module('myApp', []);
                //controller控制器 siteCtrl是控制器名字;
            app.controller('siteCtrl', function($scope, $http) {
                $http({
                //根据需要修改不同的GET或POST请求方式
                method: 'GET',
                url: '../text/1.text'
                }).then(function successCallback(response) {
                    //数据请求成功执行的代码
                $scope.names = response.data.sites;

                }, function errorCallback(response) {
                    // 请求失败执行代码
                });

            });

angular中的get方法：

            var app = angular.module('myApp', []);
                app.controller('siteCtrl', function($scope, $http) {
                  $http.get("url地址")
                    //data后面跟着是个json数组，  i sites是数组的name
                //***json数组一定要用“双引号”
                  .then(function (response) {$scope.names = response.data.sites;});
            });

angular中的post方法：

            var app = angular.module('myApp', []);
                app.controller('siteCtrl', function($scope, $http) {
                  $http.post("url地址")
                    //data后面跟着是个json数组，  i sites是数组的name
                //***json数组一定要用“双引号”
                  .then(function (response) {$scope.names = response.data.sites;});
            });
