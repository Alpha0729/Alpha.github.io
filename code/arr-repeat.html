JS数组排序

1.利用js内置的sort对象进行排序：

例如一组数组：8,5,7,6,9,0,1

从小到大排序：

注：默认是从小到大,a-b，相反b-a

sort的JS代码：

                window.onload=function(){

                function sort(){

                //用sort()方法进行排序；
                return arr.sort(function(a,b){

                return a-b;

                })

                };
2.换位排序：

1.我为什么叫它换位排序，因为它确实是在换位：

例如一组数组：8,5,-1,7,6,9,-1,0,1

从小到大排序：

JS代码：

        var oBtnA=document.getElementById('btn-1');
        var oCont2=document.getElementById('cont2');
        var oSpan2=document.getElementById('span2');
        var oBtnB=document.getElementById('btn-2');
        var oS2=oSpan2.innerText;
        var arr2=oS2.split(',');
        function fn(T){
            var max=arr2[T];
            for(var j=T;j< arr2.length;j++){
                //小于号是从小到大
                if(arr2[j]<=max){
                    //找到数组中的最小书的下角标;
                    max=arr2[j];
                    //8,5,-1,7,6,9,0,-1,1
                    var w=j;
                    /*切记return 只能放在函数最底层*/
                }
            }
            //返回出去下角标；
            return w;
        }
        function Example(){
            for(var i=0;i< arr2.length;i++){
                //拿回来的下角标赋给 n
                var n=fn(i);
                //让arr的第i++个先赋给m；
                var m=arr2[i];
                //然后把数组中的第n个，也就是最那回来的n赋给arr的第i++个；
                arr2[i]=arr2[n];
                //在把刚才的m给arr的第n个，这样刚好做到互换；
                arr2[n]=m;
            }
            return arr2;
        };
                oBtnA.onclick= function () {
            oCont2.innerHTML="从小到大排序："+''+ Example()+'';
        };
