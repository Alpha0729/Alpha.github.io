用JS使页面font-size：按比例显示：

            (function(win,doc){
                //  resize就是onresize事件;  当页面窗口没改变时触发该事件
                win.addEventListener('resize',change,false);
                function change(){
                    //以320屏幕为基础  也就是屏幕的宽度乘以20  是html的初始字体大小
                    doc.documentElement.style.fontSize=doc.documentElement.clientWidth/320*20+'px';
             }
                change();
            })(window,document)

            *****使用rem的时候移动要除以20  因为默认字体是20px;
